## <a name="example"></a><span data-ttu-id="90f4e-107">Örnek</span><span class="sxs-lookup"><span data-stu-id="90f4e-107">Example</span></span>

```csharp
{
    event MyEvent Clicked;
}

    event MyEvent ITest.Clicked;  // CS0071

        {
            clicked += value;
        }
        remove
        {
            clicked -= value;
        }
    }
    */
    public static void Main() { }
}
```
